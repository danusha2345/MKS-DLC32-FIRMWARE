#include <Arduino.h>
#include <WiFi.h>

#include <Preferences.h>

Preferences preferences;

/*
21 a
20 b
10 c
9 d

6 k

struct INPUT_EVENT
{
  uint8_t char_code;
  int16_t value;
};

struct BTN
{
  uint8_t btn_pin = 0;
  uint8_t char_code = 0;

  bool state = false;
  bool real_state = false;
  uint64_t change_time = 0;

  BTN(uint8_t _btn_pin, char _char_code) : 
    btn_pin(_btn_pin), char_code(_char_code)
  {

  }
};

BTN buttons[] =
{
  BTN {21, 'A'},
  BTN {20, 'B'},
  BTN {10, 'C'},
  BTN {9, 'D'},
  BTN {8, 'E'},
  BTN {7, 'F'},
  BTN {6, 'K'}
};

#define BUTTONS_COUNT (sizeof(buttons) / sizeof(buttons[0]))

#define EVERY_N_MICROS(loop_name, time) \ 
  static uint64_t ___##loop_name = 0; \
  for(;___##loop_name < cur_time;___##loop_name = cur_time + time)

#define X_AXIS_PIN 0
#define Y_AXIS_PIN 1

#define AXIS_MIN_VAL 5
#define AXIS_MAX_VAL 2840

int32_t axis_x_center;
int32_t axis_y_center;

int32_t prev_x_axis_val = 0;
int32_t prev_y_axis_val = 0;

void send_changes(INPUT_EVENT event)
{
  Serial0.write((const uint8_t*)&event, sizeof(event));

  Serial.printf("INPUT_EVENT %i %i\n", (int)event.char_code, (int)event.value);
}

void interrupt()
{
  auto now_us = micros();

  for(auto i = 0; i < BUTTONS_COUNT; i++)
  {
    auto button = &buttons[i];

    auto state = digitalRead(button->btn_pin) == LOW;


    if(state != button->state)
    {
      auto last_change = (now_us - button->change_time);

      if(last_change > 1000)
      {
        send_changes(INPUT_EVENT{button->char_code, (int16_t)state});

        button->real_state = state;
      }

      button->change_time = now_us;
      button->state = state;
    }
  }
}

bool is_button_pressed(char button)
{
  for(auto i = 0; i < BUTTONS_COUNT; i++)
  {
    if(buttons[i].char_code == button)
      return buttons[i].real_state;
  }

  return false;
}

void setup() 
{
  Serial.begin(115200);

  Serial0.begin(9600, SERIAL_8N1, -1, A5);

  WiFi.mode(WIFI_STA);

  if (esp_now_init() != ESP_OK) 
  {
    Serial.println("Error initializing ESP-NOW");
  }

  for(auto i = 0; i < BUTTONS_COUNT; i++)
  {
    auto button = &buttons[i];

    pinMode(button->btn_pin, INPUT);

    attachInterrupt(digitalPinToInterrupt(button->btn_pin), interrupt, CHANGE);
  }

  pinMode(X_AXIS_PIN, ANALOG);
  pinMode(Y_AXIS_PIN, ANALOG);

  preferences.begin("joystick", false);

  axis_x_center = preferences.getInt("axis_x_center", 1400);
  axis_y_center = preferences.getInt("axis_y_center", 1400);
}

int32_t read_avg(uint16_t pin, uint32_t read_count = 5)
{
  int32_t avg = 0;

  for(auto i = 0; i < read_count; i++)
  {
    avg = avg + analogRead(pin);
  }

  return avg / read_count;
}

int32_t read_axis(uint16_t pin, int32_t min, int32_t mid, int32_t max)
{
  int32_t val = (read_avg(pin) - mid);

  float res;

  if(val < 0)
  {
    res = (val - min) * 100 / (mid - min);
  }
  else
  {
    res = val * 100 / (max - mid);
  }

  return res;
}

bool send_axis_event(int32_t prev_val, int32_t val)
{
  return abs(prev_x_axis_val - x_val) > 3);
}

void loop() 
{
  delay(1);

  auto cur_time = micros();

  EVERY_N_MICROS(read_joystick, 3000)
  {
    auto x_val = read_axis(X_AXIS_PIN, 5, axis_x_center, 2235);
    auto y_val = read_axis(Y_AXIS_PIN, 5, axis_y_center, 2235);

    if(abs(prev_x_axis_val - x_val) > 3)
    {
      send_changes(INPUT_EVENT{'X', (int16_t)x_val});

      prev_x_axis_val = x_val;
    }

    if(abs(prev_y_axis_val - y_val) > 3)
    {
      send_changes(INPUT_EVENT{'Y', (int16_t)y_val});

      prev_y_axis_val = y_val;
    }
  }
  
  static uint32_t last_preferences_change = 0;
  
  if(is_button_pressed('A') && is_button_pressed('B') && is_button_pressed('C') && is_button_pressed('D'))
  {
    if(millis() - last_preferences_change > 3000)
    {
      axis_x_center = read_avg(X_AXIS_PIN);
      axis_y_center = read_avg(Y_AXIS_PIN);

      preferences.putInt("axis_x_center", axis_x_center);
      preferences.putInt("axis_y_center", axis_y_center);

      last_preferences_change = millis();

      Serial.printf("save axis_x_center=%i, axis_y_center=%i\n",
        (int)axis_x_center, (int)axis_y_center);
    }
  }
}
